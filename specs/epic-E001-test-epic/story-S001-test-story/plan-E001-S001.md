## 3. 接口契约

### 3.1 StoryStateManager API

```typescript
/**
 * Story 状态管理器
 * 负责管理 Story 在四个阶段的生命周期
 */
class StoryStateManager {
  /**
   * 构造函数
   * @param storyId - Story 唯一标识符
   */
  constructor(storyId: string): StoryStateManager;

  /**
   * 获取当前状态快照
   * @returns 当前 Story 状态的深拷贝
   */
  getCurrentState(): StoryState;

  /**
   * 获取当前阶段
   * @returns 当前所处的 StoryStage
   */
  getCurrentStage(): StoryStage;

  /**
   * 推进到下一阶段
   * @returns Result<StoryState, StageTransitionError>
   * @precondition 当前阶段已标记完成
   * @postcondition 状态更新为下一阶段，记录审计日志
   */
  advanceStage(): Result<StoryState, StageTransitionError>;

  /**
   * 标记当前阶段完成
   * @param result - 阶段执行结果
   * @returns Result<StoryState, ValidationError>
   * @precondition 存在当前阶段
   * @postcondition 当前阶段标记为完成，记录审计日志
   */
  completeStage(result: 'success' | 'failed' | 'skipped'): Result<StoryState, ValidationError>;

  /**
   * 检查是否可以推进
   * @returns true 如果当前阶段已完成且存在下一阶段
   */
  canAdvance(): boolean;
}
```

### 3.2 StageValidator API

```typescript
/**
 * 阶段验证器
 * 负责验证阶段准入条件
 */
class StageValidator {
  /**
   * 验证进入目标阶段的条件
   * @param context - 验证上下文
   * @returns ValidationResult 包含验证结果和错误信息
   */
  validateEntry(context: ValidationContext): ValidationResult;

  /**
   * 获取阶段的前置阶段
   * @param stage - 目标阶段
   * @returns 前置阶段或 null
   */
  getPrerequisiteStage(stage: StoryStage): StoryStage | null;
}
```

### 3.3 AuditLogger API

```typescript
/**
 * 审计日志记录器
 * 负责记录所有阶段生命周期事件
 */
class AuditLogger {
  /**
   * 记录阶段进入事件
   * @param storyId - Story ID
   * @param stage - 进入的阶段
   * @param input - 可选输入数据
   * @param executor - 执行者标识
   * @returns 创建的日志条目
   */
  logStageEntered(
    storyId: string,
    stage: StoryStage,
    input?: Record<string, unknown>,
    executor?: string
  ): AuditLogEntry;

  /**
   * 记录阶段完成事件
   * @param storyId - Story ID
   * @param stage - 完成的阶段
   * @param result - 执行结果
   * @param output - 可选输出数据
   * @param executor - 执行者标识
   * @returns 创建的日志条目
   */
  logStageCompleted(
    storyId: string,
    stage: StoryStage,
    result: 'success' | 'failed' | 'skipped',
    output?: Record<string, unknown>,
    executor?: string
  ): AuditLogEntry;

  /**
   * 按查询条件检索日志
   * @param query - 查询条件
   * @returns 匹配的日志条目数组
   */
  queryLogs(query: LogQuery): AuditLogEntry[];
}
```

### 3.4 错误类型定义

```typescript
// 基础验证错误
class ValidationError extends Error {
  code: string;           // 错误代码 (如 'STAGE_NOT_COMPLETED')
  stage: StoryStage;      // 相关阶段
  details: Record<string, unknown>;  // 额外上下文
}

// 阶段转换错误
class StageTransitionError extends ValidationError {
  fromStage: StoryStage;  // 源阶段
  toStage: StoryStage;    // 目标阶段
}

// 错误代码规范
const ErrorCodes = {
  STAGE_NOT_COMPLETED: 'STAGE_NOT_COMPLETED',
  INVALID_TRANSITION: 'INVALID_TRANSITION',
  SKIP_NOT_ALLOWED: 'SKIP_NOT_ALLOWED',
  REVERSE_NOT_ALLOWED: 'REVERSE_NOT_ALLOWED',
  ALREADY_FINAL_STAGE: 'ALREADY_FINAL_STAGE',
  PREREQUISITE_NOT_MET: 'PREREQUISITE_NOT_MET'
} as const;
```
